# --solutions--

```swift
class Animal {
    var gender: String?
    var legs = 4

    func description() {
        if let gender = self.gender { print(gender) }
        print(self.legs)
    }
}
let dog = Animal()
dog.gender = "female"
dog.description()
```

# --output--

female
4
