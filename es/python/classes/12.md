# --solutions--

```python
class Animal:
    def __init__(self, tipo):
        self.tipo = tipo
class Perro(Animal):
    def __init__(self, raza):
        Animal.__init__(self, 'mamifero')
            self.raza = raza
perro = Perro('Chihuahua')
print(perro.tipo)
print(perro.raza)
```
