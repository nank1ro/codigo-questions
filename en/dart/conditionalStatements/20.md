```dart
String ageGroup(int age) {
  if (age < 13) {
    return 'child';
  } else if (age < 18) {
    return 'teenager';
  } else if (age < 65) {
    return 'adult';
  } else {
    return 'senior';
  }
}

void main() {
  print(ageGroup(30));
}
```
