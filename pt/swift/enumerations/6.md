enum FlightStatus {
    case ontime, cancelled, delayed
}
let status = FlightStatus.cancelled
switch status {
    case .ontime:
        print("The flight is ontime")
    case .cancelled:
        print("The flight is cancelled")
    case .delayed:
        print("The flight is delayed")
}
