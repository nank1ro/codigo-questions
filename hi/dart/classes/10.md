
```dart
  test('test2', () {
    final c = Counter();
    c.increment();
    c.increment();
    c.reset();
    expect(c.count, 0, reason: '--err-t2--');
  });
```

# --solutions--

```dart
class Counter {
  int count = 0;

  void increment() {
    count++;
  }

  void reset() {
    count = 0;
  }
}
```
